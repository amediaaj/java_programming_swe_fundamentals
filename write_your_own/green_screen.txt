function topRightCorner(cornerWidth, cornerHeight, image, red, green, blue){
    var width = image.getWidth();
    for(var pixel of image.values()){
        if(pixel.getY()< cornerHeight){
            if(pixel.getX() > width - cornerWidth){
                pixel.setRed(red);
                pixel.setGreen(green);
                pixel.setBlue(blue);
            }
        }
    }
    return image;
}
